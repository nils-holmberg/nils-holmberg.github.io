== docx, markdown (201105)

== kaggle api (200918)

pip install kaggle --user
# i.e. not system wide install
#sudo pip3 install kaggle

pip3 show kaggle
# download kaggle api key, define location
export KAGGLE_CONFIG_DIR=/home/sol-nhl/doc/env/kaggle
# set perms
chmod 600 /home/sol-nhl/doc/env/kaggle/kaggle.json
# interact with api
kaggle kernels list -s scom
# check config
kaggle config view
# set path, not needed..
kaggle config set -n path -v /home/sol-nhl/res/projects/scom-tour/src
# initialize local dir
kaggle kernels init -p /home/sol-nhl/res/projects/scom-tour/src
# specify kernel id and code file
vim src/kernel-metadata.json 
# pull remote kernel
kaggle kernels pull -k nilsholmberg/scom-tour -p /home/sol-nhl/res/projects/scom-tour/src -m
# update locally
cat src/some-code | jq
vim src/some-code 
# push chages to remote
