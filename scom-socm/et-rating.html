getpvals_simple <- function(m) {
#coefs
coefs <- data.frame(coef(summary(m)))
# use normal distribution to approximate p-value
coefs$p.z <- 2 * (1 - pnorm(abs(coefs$t.value)))
#round
mp = round(coefs, 3)
#
return(mp)
}
